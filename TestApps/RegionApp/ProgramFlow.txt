--- CityListEvents.js ---
DocumentClick()
  If clicked on cityTable row then
    Call SelectRow()
      If event source is id:cityTable then
        Get clicked row.
        Set var:CityRowIndex to clicked row index.
        Clear highlight from previous row hiddenForm.cityRowIndex.
        Highlight clicked row.
        Call SetRowFormValues()
          If event source is id:cityTable then
            Set hiddenForm.cityRowIndex = CityRowIndex;
            Set hiddenForm.cityID = ID (Value of first column in row).
          
DeleteClick()
  Calls DoAction("Delete")
EditClick()
  Calls DoAction("Update")
NewClick()
  Calls DoAction("Add")

  DoAction(action);
    hiddenForm.cityID is not set so no selected row.
      Skip submit for "Delete" and "Update".
    Set input text hiddenForm.listAction.value.
    Submit(hiddenForm) to CityDetail.php
